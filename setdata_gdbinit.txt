# 1-3-4
define setData1
    set $nI=0

    while ($nI < 5)
    set iniAlmHost[$nI].kansiHostNo=$nI+1

      if ($nI == 0)
        set iniAlmHost[$nI].redFCnt=5
        set iniAlmHost[$nI].redCnt=5
      end

      if ($nI == 1)
        set iniAlmHost[$nI].yelFCnt=5
        set iniAlmHost[$nI].yelCnt=5
      end

      if ($nI == 2)
        set iniAlmHost[$nI].redFCnt=0
        set iniAlmHost[$nI].redCnt=5
      end

      if ($nI == 3)
        set iniAlmHost[$nI].yelFCnt=0
        set iniAlmHost[$nI].yelCnt=5
      end

      if ($nI == 4)
        set iniAlmHost[$nI].grnCnt=5
      end

      set $nI=$nI+1
    end

    set iniAlmHost[$nI].kansiHostNo=0
    p iniAlmHost
end

        set iniAlmHost[$nI].redFCnt=0
        set iniAlmHost[$nI].redCnt=5
